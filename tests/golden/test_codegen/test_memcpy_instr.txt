#include "bar.h"

/* Required by memcpy(n,dst,src) */
#include <string.h>
// bar(
//     n : size,
//     dst : f32[n] @DRAM,
//     src : f32[n] @DRAM
// )
void bar( void *ctxt, int_fast32_t n, float* dst, const float* src ) {
memcpy(dst, src, (n + 0) * sizeof(float));
}


/* relying on the following instruction..."
memcpy(n,dst,src)
memcpy({dst}, {src}, {n} * sizeof(float));
*/
