def svmla(N: size, C: f32[N] @ DRAM, A: f32[N] @ DRAM, B: f32 @ DRAM):
    assert N == 16
    C_reg: f32[N] @ SVE_VLS
    svld1_f32(N, C_reg[0:N], C[0:N])
    A_reg: f32[N] @ SVE_VLS
    svld1_f32(N, A_reg[0:N], A[0:N])
    for i in seq(0, N):
        C_reg[i] += A_reg[i] * B
    for i0 in seq(0, N):
        C[i0] = C_reg[i0]
