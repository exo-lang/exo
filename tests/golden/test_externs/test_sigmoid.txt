float sigmoid(float x) {
    return 1 / (1 + exp(-x));
}

// foo(
//     x : f32[16] @DRAM,
//     y : f32[16] @DRAM
// )
void foo( void *ctxt, const float* x, float* y ) {
for (int_fast32_t i = 0; i < 16; i++) {
  y[i] = sigmoid((float)(x[i] + y[i]));
}
}
