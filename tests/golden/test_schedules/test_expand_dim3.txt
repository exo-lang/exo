// foo(
//     n : size,
//     m : size,
//     x : i8 @DRAM
// )
void foo( void *ctxt, int_fast32_t n, int_fast32_t m, int8_t* x ) {
for (int_fast32_t i = 0; i < n; i++) {
  for (int_fast32_t j = 0; j < m; j++) {
    ; // NO-OP
  }
}
for (int_fast32_t i = 0; i < n; i++) {
  int8_t *a = (int8_t*) malloc(n * sizeof(*a));
  for (int_fast32_t j = 0; j < m; j++) {
    *x = a[i];
  }
  free(a);
}
for (int_fast32_t i = 0; i < n; i++) {
  for (int_fast32_t j = 0; j < m; j++) {
    ; // NO-OP
  }
}
