inline void
exo_CudaInline_test_CudaWarps_in_wg_golden::exo_Cuda0_proc_CudaWarps_in_wg::exo_cudaLaunch(
    cudaStream_t exo_cudaStream,
    const exo_DeviceArgs& exo_deviceArgs)
{
  namespace exo_CudaUtil = exo_CudaUtil_test_CudaWarps_in_wg_golden;
  cudaFuncSetAttribute(exo_deviceFunction0_proc_CudaWarps_in_wg, cudaFuncAttributeMaxDynamicSharedMemorySize, exo_smemBytes);
  // TODO how expensive is it to query this every time?
  int exo_cudaDevice;
  cudaGetDevice(&exo_cudaDevice);
  int exo_SMs;
  cudaDeviceGetAttribute(&exo_SMs, cudaDevAttrMultiProcessorCount, exo_cudaDevice);
  const unsigned exo_gridDim = (unsigned(exo_SMs) & ~(exo_clusterDim - 1)) * 1u;

  cudaLaunchConfig_t exo_launchConfig = {};
  exo_launchConfig.gridDim = dim3(exo_gridDim, 1, 1);
  exo_launchConfig.blockDim = dim3(exo_blockDim, 1, 1);
  exo_launchConfig.dynamicSmemBytes = exo_smemBytes;
  exo_launchConfig.stream = exo_cudaStream;

  cudaLaunchKernelEx(&exo_launchConfig, exo_deviceFunction0_proc_CudaWarps_in_wg, exo_deviceArgs);

  exo_excut_flush_device_log(
      exo_cudaStream, exo_gridDim, exo_blockDim,
      exo_CudaUtil::exo_excut_str_id_count, exo_CudaUtil::exo_excut_str_table,
      1, &exo_FILE());
}

__device__ __forceinline__ void
exo_CudaInline_test_CudaWarps_in_wg_golden::exo_Cuda0_proc_CudaWarps_in_wg::exo_deviceTask(
    char* exo_smem,
    exo_SyncState& exo_syncState,
    const exo_DeviceArgs& exo_deviceArgs,
    exo_Task exo_task,
    exo_ExcutThreadLog exo_excutLog)
{
  namespace exo_CudaUtil = exo_CudaUtil_test_CudaWarps_in_wg_golden;
  // cuda_threads(0, 4, unit=cuda_warpgroup)
  if ([[maybe_unused]] int exo_128thr_wg = (threadIdx.x / 128); 1) {
    // cuda_threads(0, 128, unit=cuda_thread)
    if ([[maybe_unused]] int exo_1thr_t = (threadIdx.x % 128); 1) {
      exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
      exo_excutLog.log_u32_arg(0);
      exo_excutLog.log_u32_arg(exo_128thr_wg);
      exo_excutLog.log_u32_arg(exo_1thr_t);
    }
    // CudaWarps(2, 4)
    if (int tmp_1 = (threadIdx.x % 128); tmp_1 >= 64) {
      // cuda_threads(0, 64, unit=cuda_thread)
      if ([[maybe_unused]] int exo_1thr_t = (threadIdx.x % 128 - 64); 1) {
        exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
        exo_excutLog.log_u32_arg(1);
        exo_excutLog.log_u32_arg(exo_128thr_wg);
        exo_excutLog.log_u32_arg(exo_1thr_t);
      }
      // CudaWarps(1, 2)
      if (int tmp_2 = (threadIdx.x % 128 - 64); tmp_2 >= 32) {
        // cuda_threads(0, 32, unit=cuda_thread)
        if ([[maybe_unused]] int exo_1thr_t = (threadIdx.x % 128 - 96); 1) {
          exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
          exo_excutLog.log_u32_arg(2);
          exo_excutLog.log_u32_arg(exo_128thr_wg);
          exo_excutLog.log_u32_arg(exo_1thr_t);
        }
      }
    }
  }
  // CudaWarps(8, 16)
  if (int tmp_1 = threadIdx.x; tmp_1 >= 256) {
    // cuda_threads(0, 2, unit=cuda_warpgroup)
    if ([[maybe_unused]] int exo_128thr_wg = ((threadIdx.x - 256) / 128); 1) {
      // CudaWarps(0, 2)
      if (int tmp_2 = (threadIdx.x % 128); tmp_2 < 64) {
        // cuda_threads(0, 64, unit=cuda_thread)
        if ([[maybe_unused]] int exo_1thr_t = (threadIdx.x % 128); 1) {
          exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
          exo_excutLog.log_u32_arg(3);
          exo_excutLog.log_u32_arg(exo_128thr_wg);
          exo_excutLog.log_u32_arg(exo_1thr_t);
        }
      }
      // CudaWarps(2, 4)
      if (int tmp_2 = (threadIdx.x % 128); tmp_2 >= 64) {
        // cuda_threads(0, 64, unit=cuda_thread)
        if ([[maybe_unused]] int exo_1thr_t = (threadIdx.x % 128 - 64); 1) {
          exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
          exo_excutLog.log_u32_arg(4);
          exo_excutLog.log_u32_arg(exo_128thr_wg);
          exo_excutLog.log_u32_arg(exo_1thr_t);
        }
        // cuda_threads(0, 42, unit=cuda_thread)
        if (int exo_1thr_t = (threadIdx.x % 128 - 64); exo_1thr_t < 42) {
          exo_excutLog.log_action(EXO_EXCUT_STR_ID(excut_trace_3index), 0, __LINE__);
          exo_excutLog.log_u32_arg(5);
          exo_excutLog.log_u32_arg(exo_128thr_wg);
          exo_excutLog.log_u32_arg(exo_1thr_t);
        }
      }
    }
  }
}
__device__ __forceinline__ void
exo_CudaInline_test_CudaWarps_in_wg_golden::exo_Cuda0_proc_CudaWarps_in_wg::exo_deviceMainLoop(
    char* exo_smem,
    const exo_DeviceArgs& exo_deviceArgs,
    exo_ExcutThreadLog exo_excutLog)
{
  namespace exo_CudaUtil = exo_CudaUtil_test_CudaWarps_in_wg_golden;
  exo_SyncState exo_syncState{};
  unsigned exo_taskIndex = 0;
  int32_t nreg;
  nreg = ((int32_t) 0);
  if (nreg == ((int32_t) 0)) {
    if (int tmp = threadIdx.x; tmp >= 0 && tmp < 512) {
      for (int exo_task_task = 0; exo_task_task < 3; exo_task_task++) {
        if (exo_taskIndex++ % (gridDim.x / exo_clusterDim) == blockIdx.x / exo_clusterDim) {
            exo_deviceTask(exo_smem, exo_syncState, exo_deviceArgs,
                (struct exo_Task) { exo_task_task },
                exo_excutLog);
        }
      }
    }
  }
}
